cmake_minimum_required(VERSION 3.15)
project(pzep-gui LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED OFF)

# Manual paths - NO vcpkg toolchain
set(RAYLIB_ROOT "C:/vcpkg/included/x64-windows")
set(VCPKG_ROOT "C:/vcpkg/included/x64-windows")
set(ZEP_ROOT "C:/Users/casse/github/zep")

# Include directories  
include_directories(
    "${ZEP_ROOT}/include"
    "${ZEP_ROOT}/include/zep"
    "${VCPKG_ROOT}/include"
)

# Library paths  
set(RAYLIB_LIB "C:/vcpkg/installed/x64-windows/lib/raylib.lib")
set(ZEP_LIB "${ZEP_ROOT}/build-win/lib/Release/Zep.lib")

# Source files
set(SOURCES
    pzep_gui.cpp
    "${ZEP_ROOT}/src/raylib/display_raylib.cpp"
)

# Create executable
add_executable(pzep_gui ${SOURCES})

# Link libraries
target_link_libraries(pzep_gui PRIVATE 
    "${RAYLIB_LIB}"
    "${ZEP_LIB}"
    opengl32
    winmm  
    gdi32
)

# Preprocessor definitions
target_compile_definitions(pzep_gui PRIVATE 
    _CRT_SECURE_NO_WARNINGS
    PLATFORM_DESKTOP
)
